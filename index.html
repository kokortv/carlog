📖 Настройка Apps Script
1. Создайте Google Таблицу
Перейдите на sheets.google.com и создайте новую таблицу.
Скопируйте ID из URL: /spreadsheets/d/ВАШ_ID/edit

2. Откройте Apps Script
В таблице: Расширения → Apps Script

3. Вставьте этот код
Замените ВАШ_ID на ID вашей таблицы:
const SHEET_ID = "ВАШ_ID_ТАБЛИЦЫ"; function doPost(e) { try { const d = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.openById(SHEET_ID); const sh = ss.getSheetByName(d.sheet) || ss.insertSheet(d.sheet); if (d.action === "append") { sh.appendRow(d.row); } else if (d.action === "update") { const rows = sh.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (String(rows[i][0]) === String(d.id)) { d.row.forEach((v, j) => sh.getRange(i + 1, j + 1).setValue(v)); break; } } } else if (d.action === "delete") { const rows = sh.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (String(rows[i][0]) === String(d.id)) { sh.deleteRow(i + 1); break; } } } else if (d.action === "init") { initHeaders(ss); } return ok(); } catch(err) { return ContentService .createTextOutput( JSON.stringify({ok:false,e:err.message})) .setMimeType(ContentService.MimeType.JSON); } } function initHeaders(ss) { const cfg = { Cars: ["ID","Brand","Model","Year", "Mileage","VIN","Color","CreatedAt"], Fuel: ["ID","CarID","Date","Mileage", "Liters","PricePerLiter","Currency", "Total","Note","CreatedAt"], Service: ["ID","CarID","Date","ServiceType", "Mileage","Cost","Currency", "Note","CreatedAt"] }; for (const [name, hdrs] of Object.entries(cfg)) { let s = ss.getSheetByName(name); if (!s) { s = ss.insertSheet(name); s.appendRow(hdrs); } else if (s.getLastRow() === 0) s.appendRow(hdrs); } } function ok() { return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); }
4. Разверните как веб-приложение
Нажмите Развернуть → Новое развёртывание
• Тип: Веб-приложение
• Выполнять как: Я
• Доступ: Все
Скопируйте URL и вставьте в настройки CarLog.

⚠ Важно
При каждом изменении кода нужно создавать новое развёртывание (не обновлять существующее), иначе URL не обновится.

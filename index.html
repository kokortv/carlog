Шаг 1. Создайте Google Таблицу
Откройте sheets.google.com и создайте новую таблицу. Листы создадутся автоматически.

Шаг 2. Apps Script
В таблице: Расширения → Apps Script
Удалите всё и вставьте код ниже, заменив ВАШ_ID:
const SPREADSHEET_ID = "ВАШ_ID_ТАБЛИЦЫ"; // ID находится в URL таблицы: // docs.google.com/spreadsheets/d/ВАШ_ID/edit // ← Обязательно! Обрабатываем GET-запросы (проверка деплоя) function doGet(e) { return respond({ok: true, msg: "CarLog Script is running"}); } function doPost(e) { // Защита от пустого тела (GET-запросы и пинги) if (!e || !e.postData || !e.postData.contents) { return respond({ok: false, msg: "No data received"}); } try { const d = JSON.parse(e.postData.contents); if (!d || !d.action) { return respond({ok: false, msg: "Invalid request"}); } const ss = SpreadsheetApp.openById(SPREADSHEET_ID); // Инициализация — создаём листы с заголовками if (d.action === "init") { setupSheets(ss); return respond({ok: true, msg: "Sheets ready"}); } // Для остальных действий нужен лист if (!d.sheet) return respond({ok: false, msg: "No sheet specified"}); const sheet = ss.getSheetByName(d.sheet); if (!sheet) return respond({ok: false, msg: "Sheet not found: " + d.sheet}); if (d.action === "append") { if (!d.row || !Array.isArray(d.row)) { return respond({ok: false, msg: "No row data"}); } sheet.appendRow(d.row); } else if (d.action === "update") { if (!d.id || !d.row) return respond({ok: false, msg: "Missing id or row"}); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (String(data[i][0]) === String(d.id)) { d.row.forEach((val, col) => sheet.getRange(i + 1, col + 1).setValue(val) ); break; } } } else if (d.action === "delete") { if (!d.id) return respond({ok: false, msg: "Missing id"}); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (String(data[i][0]) === String(d.id)) { sheet.deleteRow(i + 1); break; } } } return respond({ok: true}); } catch(err) { return respond({ok: false, msg: err.message}); } } function setupSheets(ss) { const config = { "Cars": ["ID","Brand","Model","Year","Mileage","VIN","Color","CreatedAt"], "Fuel": ["ID","CarID","Date","Mileage","Liters","PricePerLiter","Currency","Total","Note","CreatedAt"], "Service": ["ID","CarID","Date","ServiceType","Mileage","Cost","Currency","Note","CreatedAt"] }; for (const [name, headers] of Object.entries(config)) { let sheet = ss.getSheetByName(name); if (!sheet) { sheet = ss.insertSheet(name); } // Добавляем заголовки только если лист пустой if (sheet.getLastRow() === 0) { sheet.appendRow(headers); sheet.getRange(1, 1, 1, headers.length) .setFontWeight("bold") .setBackground("#1a73e8") .setFontColor("#ffffff"); sheet.setFrozenRows(1); } } } function respond(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
Шаг 3. Разверните
Нажмите Развернуть → Новое развёртывание
• Тип: Веб-приложение
• Выполнять как: Я
• Доступ: Все
Скопируйте URL и вставьте в поле выше.

⚠ Если в таблице появился пустой лист без названия — просто удалите его. Это происходило из-за старой версии скрипта. Новый код это исправляет.
Шаг 4. Вшейте URL — один раз для всех устройств
URL хранится в памяти браузера — значит на телефоне его придётся вводить заново. Чтобы этого избежать:
Откройте index.html в текстовом редакторе, найдите строку:
const HARDCODED_SCRIPT_URL = "";
Вставьте ваш URL между кавычками:
const HARDCODED_SCRIPT_URL = "https://script.google.com/...";
Сохраните файл и загрузите на хостинг. Теперь URL работает на любом устройстве — вводить ничего не нужно.
